## Task 2: Creative Effect — Implementation

**Files:**

**Step 1: Implement effect-creative.js**

Follow the `effect-showcase.js` pattern. Key differences:
- `prev`/`next`/`active` config objects define target transforms
- Resolve `'100%'` string values to `slideSize * (percentage / 100)`
- Interpolate all properties linearly based on `normalizedOffset`
- For offset < 0 (prev side): interpolate toward `prev` config
- For offset > 0 (next side): interpolate toward `next` config
- `active` defaults to `{ translate: [0,0,0], rotate: [0,0,0], scale: 1, opacity: 1 }`
- Build transform string: `translateX() translateY() translateZ() rotateX() rotateY() rotateZ() scale()`
- When `perspective: true`, set `slider.trackEl.style.perspective = '1200px'`
- Use CSS transitions (like showcase), NOT rAF (creative doesn't have coverflow's overflow:hidden issue)
- Override `setTranslate`, `setTransition`, `getComputedTranslate`, and optionally `loopFix` (if loop + centeredSlides)

Key helper function:

```js
function resolveValue(val, slideSize) {
  if (typeof val === 'string' && val.endsWith('%')) {
    return (parseFloat(val) / 100) * slideSize;
  }
  return Number(val) || 0;
}

function interpolateConfig(config, absOffset) {
  // Returns resolved { tx, ty, tz, rx, ry, rz, scale, opacity }
  const t = config.translate || [0, 0, 0];
  const r = config.rotate || [0, 0, 0];
  return {
    tx: resolveValue(t[0], slider.slideSize) * absOffset,
    ty: resolveValue(t[1], slider.slideSize) * absOffset,
    tz: resolveValue(t[2], slider.slideSize) * absOffset,
    rx: (r[0] || 0) * absOffset,
    ry: (r[1] || 0) * absOffset,
    rz: (r[2] || 0) * absOffset,
    scale: 1 - (1 - (config.scale ?? 1)) * absOffset,
    opacity: 1 - (1 - (config.opacity ?? 1)) * absOffset,
  };
}
```

**Step 1: Implement thumbs.js**

```js
export default function Thumbs({ slider, extendParams, on }) {
  extendParams({
    thumbs: {
      slider: null,
      slideThumbActiveClass: 'drift-thumb--active',
      autoScrollThumbs: true,
      multipleActiveThumbs: false,
    },
  });

  let thumbsSlider = null;
  const clickHandlers = [];

  function updateActiveThumb() {
    if (!thumbsSlider || thumbsSlider.destroyed) return;
    const params = slider.params.thumbs;
    const activeClass = params.slideThumbActiveClass;
    const realIndex = slider.realIndex ?? slider.activeIndex;

    // Remove active from all thumbs
    for (let i = 0; i < thumbsSlider.slides.length; i++) {
      thumbsSlider.slides[i].classList.remove(activeClass);
    }

    // Add active to matching thumb(s)
    for (let i = 0; i < thumbsSlider.slides.length; i++) {
      const thumbRealIndex = thumbsSlider._loopedSlides
        ? thumbsSlider._getRealIndex(i)
        : i;
      if (thumbRealIndex === realIndex) {
        thumbsSlider.slides[i].classList.add(activeClass);
      }
    }

    // Auto-scroll thumbs to keep active visible
    if (params.autoScrollThumbs) {
      thumbsSlider.slideTo(realIndex, slider.params.speed);
    }
  }

  function onThumbClick(index) {
    const realIndex = thumbsSlider._loopedSlides
      ? thumbsSlider._getRealIndex(index)
      : index;
    slider.slideTo(realIndex, slider.params.speed);
  }

  function init() {
    const params = slider.params.thumbs;
    if (!params || !params.slider) return;

    thumbsSlider = params.slider;

    // Bind click on each thumb
    for (let i = 0; i < thumbsSlider.slides.length; i++) {
      const handler = () => onThumbClick(i);
      thumbsSlider.slides[i].addEventListener('click', handler);
      clickHandlers.push({ el: thumbsSlider.slides[i], handler });
    }

    updateActiveThumb();
  }

  function destroy() {
    // Remove click listeners
    for (const { el, handler } of clickHandlers) {
      el.removeEventListener('click', handler);
    }
    clickHandlers.length = 0;

    // Remove active classes
    if (thumbsSlider && !thumbsSlider.destroyed) {
      const activeClass = slider.params.thumbs.slideThumbActiveClass;
      for (let i = 0; i < thumbsSlider.slides.length; i++) {
        thumbsSlider.slides[i].classList.remove(activeClass);
      }
    }

    thumbsSlider = null;
  }

  on('init', init);
  on('slideChange', updateActiveThumb);
  on('destroy', destroy);
}
```
