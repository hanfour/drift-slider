var DS = DriftSlider;
var slider = new DS.DriftSlider('.drift-slider', {
  modules: [DS.Navigation, DS.Pagination, DS.EffectCoverflow, DS.A11y],
  effect: 'coverflow',
  slidesPerView: 2.075,   // containerWidth / centerWidth
  centeredSlides: true,
  speed: 500,
  coverflowEffect: {
    cropSides: true,
    scale: 1,
    stretch: -16,
    depth: 0,
    rotate: 0,
    opacity: 0.8,
    overlay: true,
    align: 'center',
    staggerY: 0,
    visibleSides: 'both',
  },
});
