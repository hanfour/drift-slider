Source Code

js 
var DS = DriftSlider;
var slider = new DS.DriftSlider('.drift-slider', {
  modules: [DS.Navigation, DS.Pagination, DS.A11y],
  direction: 'vertical',
  slidesPerView: 1,
  speed: 400,
  grabCursor: true,
});
