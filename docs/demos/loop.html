Source Code

js 
var DS = DriftSlider;
var slider = new DS.DriftSlider('.drift-slider', {
  modules: [DS.Navigation, DS.Pagination, DS.Autoplay, DS.A11y],
  loop: true,
  speed: 400,
  grabCursor: true,
  autoplay: {
    enabled: true,
    delay: 3000,
    disableOnInteraction: false,
  },
});
