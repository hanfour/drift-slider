var DS = DriftSlider;
var slider;

function getConfig() {
  return {
    modules: [DS.Navigation, DS.Pagination, DS.A11y],
    slidesPerView: 1,
    spaceBetween: 20,
    grabCursor: true,
    speed: parseInt(document.getElementById('ctrl-speed').value),
    physics: {
      tension: parseInt(document.getElementById('ctrl-tension').value),
      friction: parseInt(document.getElementById('ctrl-friction').value),
    },
    edgeBounce: document.getElementById('ctrl-edge-bounce').checked,
  };
}

function createSlider() {
  if (slider) slider.destroy();
  var config = getConfig();
  slider = new DS.DriftSlider('.drift-slider', config);
  updateCodeOutput(config);
}

function updateCodeOutput(config) {
  var output = 'new DS.DriftSlider(\'.drift-slider\', {\n';
  output += '  speed: ' + config.speed + ',\n';
  output += '  physics: {\n';
  output += '    tension: ' + config.physics.tension + ',\n';
  output += '    friction: ' + config.physics.friction + ',\n';
  output += '  },\n';
  output += '  edgeBounce: ' + config.edgeBounce + ',\n';
  output += '  grabCursor: true,\n';
  output += '});';
  document.getElementById('code-output').textContent = output;
}

createSlider();

document.querySelectorAll('.controls input').forEach(function(input) {
  input.addEventListener('input', function() {
    document.getElementById('speed-value').textContent =
      document.getElementById('ctrl-speed').value;
    document.getElementById('tension-value').textContent =
      document.getElementById('ctrl-tension').value;
    document.getElementById('friction-value').textContent =
      document.getElementById('ctrl-friction').value;
    createSlider();
  });
});
