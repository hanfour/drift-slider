var DS = DriftSlider;
var slider = new DS.DriftSlider('.drift-slider', {
  modules: [DS.Navigation, DS.Pagination, DS.EffectCards, DS.A11y],
  effect: 'cards',
  speed: 500,
  cardsEffect: {
    mode: 'stack',
    direction: 'tl-br',
    offsetX: 30,
    offsetY: 30,
    scale: 0.92,
    opacity: 0.85,
    overlay: true,
    shadow: true,
  },
});
