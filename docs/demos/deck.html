var DS = DriftSlider;
var slider = new DS.DriftSlider('.drift-slider', {
  modules: [DS.Navigation, DS.Pagination, DS.EffectDeck, DS.A11y],
  effect: 'deck',
  speed: 500,
  deckEffect: {
    stackOrigin: 'bottom-left',
    stackScale: 0.6,
    stackVisibleCount: 3,
    perspective: 1200,
    tiltX: 5,
    tiltY: -3,
  },
});
