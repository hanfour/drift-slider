Creative Effect

Fully customisable per-slide transforms — choose a preset or define your own.



      
      
Source Code

      

        js
        
var DS = DriftSlider;
var slider = new DS.DriftSlider('.drift-slider', {
  modules: [DS.Navigation, DS.Pagination, DS.Autoplay, DS.EffectCreative, DS.A11y],
  effect: 'creative',
  loop: true,
  autoplay: { enabled: true, delay: 3000 },
  speed: 600,
  creativeEffect: {
    perspective: true,
    prev: { translate: ['-100%', 0, 0], rotate: [0, -30, 0] },
    next: { translate: ['100%', 0, 0], rotate: [0, 30, 0] },
  },
});
