js 
$(function() {
  $('.drift-slider').driftSlider({
    slidesPerView: 1,
    loop: true,
    navigation: true,
    pagination: { type: 'bullets' },
    autoplay: { enabled: true, delay: 4000 },
    grabCursor: true,
  });
});
