雙向跑馬燈

Two ticker rows scrolling in opposite directions at different speeds — a popular visual pattern for showcasing categories or tags.

var DS = DriftSlider;

// Top row — normal direction
var top = new DS.DriftSlider('#ticker-top', {
  modules: [DS.Autoplay],
  slidesPerView: 5,
  spaceBetween: 16,
  loop: true,
  speed: 3000,
  grabCursor: true,
  autoplay: {
    enabled: true,
    ticker: true,
    tickerSpeed: 1,
    pauseOnMouseEnter: true,
  },
});

// Bottom row — reverse direction, slightly slower
var bottom = new DS.DriftSlider('#ticker-bottom', {
  modules: [DS.Autoplay],
  slidesPerView: 5,
  spaceBetween: 16,
  loop: true,
  speed: 3000,
  grabCursor: true,
  autoplay: {
    enabled: true,
    ticker: true,
    tickerSpeed: 0.7,
    reverseDirection: true,
    pauseOnMouseEnter: true,
  },
});
